\begin{frame}
	\frametitle{Event probability}
	
	\begin{columns}[c]
		
		\column{.35\textwidth}
		
		\begin{itemize}
			\item runtime in relation to event probability
			\begin{itemize}
				\item insert
				\item mutate
				\item crossover
			\end{itemize}
			\item trade-off
			\begin{itemize}
				\item fitness evaluation
				\item event occurrence
			\end{itemize}
		\end{itemize}
		
		\column{.60\textwidth}
		\begin{equation}
			P_{adj}(e) = \frac{n_1 P_1(e) + n_2 P_2(e)}{n_1 + n_2}
		\end{equation}
		
		\begin{equation}
			\hat{P}(e) = \sum_{i=k}^n \frac{n!}{i!(n-i)!} P_{adj}(e)^i (1-P_{adj}(e))^{n-i}
		\end{equation}
		
	\end{columns}
	
\end{frame}
